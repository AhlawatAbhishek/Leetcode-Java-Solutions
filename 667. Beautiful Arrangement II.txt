class Solution {
    public int[] constructArray(int n, int k) {
        int result[] = new int[n];
        int low = 1, high = k+1;
        for(int i = 0; i<k+1; i++){
            if(i % 2 == 0){
                result[i] = low++;
            }else{
                result[i] = high--;
            }
        }
        for(int i = k+1; i<n; i++){
           result[i] = i+1;
        }
        return result;
    }
}
