class Solution {
    public boolean isValid(String s) {
        Stack<Character> st = new Stack<>();
        char c[] = s.toCharArray();
        for(char ch : c){
            if(ch == '(' || ch == '{' || ch == '['){
                st.push(ch);
            }else{
                if(ch == ']'){
                    if(!st.isEmpty() && st.peek() == '['){
                        st.pop();
                    }else{
                        return false;
                    }
                }else if(ch == ')'){
                    if(!st.isEmpty() && st.peek() == '('){
                        st.pop();
                    }else{
                        return false;
                    }
                }else if(ch == '}'){
                   if(!st.isEmpty() && st.peek() == '{'){
                        st.pop();
                    }else{
                        return false;
                    }                               
                   
                }
            }
        }
        return st.isEmpty();
    }
}
