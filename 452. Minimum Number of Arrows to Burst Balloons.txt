class Solution {
    public int findMinArrowShots(int[][] points) {
        int totalArr = 1, pLen = points.length;
        Arrays.sort(points, (a, b)->Integer.compare(a[1],b[1]));
        int end = points[0][1];
        for(int indx = 1; indx<pLen; indx++){
            if(end < points[indx][0]){
                 end = points[indx][1];
                totalArr++;
            }
        }
        return totalArr;
    }
}
