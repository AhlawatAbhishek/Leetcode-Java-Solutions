/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        if(headB == null|| headA == null ){
            return null;
        }
        int sizeA = 0, sizeB = 0;
        ListNode tempA = headA, tempB = headB;
        while(tempA != null){
            tempA = tempA.next;
            sizeA++;
        }
          while(tempB != null){
            tempB= tempB.next;
            sizeB++;
        }
        int d = Math.abs(sizeA - sizeB);
        if(sizeA > sizeB){
            int x = d;
            while(x > 0){
                headA = headA.next;
                x--;
            }
        }else{
            int x = d;
            while(x > 0){
                headB = headB.next;
                x--;
            }
        }
        while(headA != null && headB != null){
            if(headA == headB){
                return headA;
            }
            headA = headA.next;
            headB = headB.next;
        }
        return null;
    }
}
