class Solution {
    public int findPoisonedDuration(int[] timeSeries, int duration) {
        int start = 0;
        int end = 0;
        int result = 0;
        
        for(int t : timeSeries ) {
            if(t>=end) {
                result += duration;
            } else {
                result += t-start;
            }
            
            start = t;
            end = t+duration;
        } 
        
        return result;
    }
}
