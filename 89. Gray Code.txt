class Solution {
    public List<Integer> grayCode(int n) {
        List<Integer> list = new ArrayList<>();
        list.add(0);
        list.add(1);
        if(n == 1){
            return list;
        }
       
        int curr = 1;
        for(int i = 2; i<=n; i++){
            curr = curr*2;
            for(int j = list.size()-1; j>=0; j--){
                list.add(list.get(j) + curr);
            }
        }
        return list;
    }
}
