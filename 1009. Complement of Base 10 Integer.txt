class Solution {
    public int bitwiseComplement(int N) {
        if(N == 1){
            return 0;
        }
        if(N == 0){
            return 1;
        }
        int result = 0;
        int power = 1;
        while(N>0)
        {
            result += (N % 2 ^ 1) * power;
            power <<= 1; //mulitply by 2;
            N >>= 1; // divide by 2
        }
        return result;
    }
}
