class Solution {
    public int[] beautifulArray(int n) {
        ArrayList<Integer> ans = new ArrayList<>();
        ans.add(1);
        while(ans.size() < n){
            ArrayList<Integer> temp = new ArrayList<>();
            for(int el : ans){
                if(2*el-1 <= n){
                    temp.add(2*el -1);
                }
            }
            for(int el : ans){
                if(2*el <= n){
                    temp.add(2*el);
                }
            }
            ans = temp;
        }
        int res[] = new int[n];
        for(int i = 0; i<n; i++){
            res[i] = ans.get(i);
        }
        return res;
    }
}
