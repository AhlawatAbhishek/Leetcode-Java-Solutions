class Solution {
    public int smallestRangeII(int[] A, int k) {
        int n = A.length;
        Arrays.sort(A);
        int res = A[n-1] - A[0];
        for(int i = 0 ; i<n-1; i++){
            int j = i + 1;
            int low = Math.min(A[0]+k, A[j]-k);
            int high = Math.max(A[n-1] -k , A[i]+k);
            res = Math.min(res, high-low);
        }
        return res;
    }
}
