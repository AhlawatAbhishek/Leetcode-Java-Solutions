class Solution {
    public String interpret(String command) { 
         String ans="";
        int n=command.length(),i;
        for(i=0;i<n;i++){
           if(command.charAt(i)=='G'){
               ans+='G';
           }else if(command.charAt(i+1)==')'){
               i++;
               ans+='o';
           }else{
               ans+='a';
               ans+='l';
               i+=3;
           }
        }    
        return ans;
    }
}
