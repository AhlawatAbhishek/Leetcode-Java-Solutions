class Solution {
    public boolean increasingTriplet(int[] nums) {
        if(nums.length<3){
            return false;
        }
        int i = Integer.MAX_VALUE, j = Integer.MAX_VALUE;
        for(int a : nums){
            if(a <= i){
                i = a;
            }else if(a <= j){
                j = a;
            }else{
                return true;
            }
        }
        return false;
    }
}
