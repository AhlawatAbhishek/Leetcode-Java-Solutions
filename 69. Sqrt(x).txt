class Solution {
    public int mySqrt(int x) {
        if(x<2) return x;
        long left = 0, right = x/2;
        long res = -1;
        while(left <= right){
            long mid = left + (right - left)/2;
            if((long)mid*mid == x){
                return (int)mid;
            }else if(mid*mid<x){
                res = mid;
                left = mid+1;
            }else{
                right = mid - 1;
            }
        }
        return (int)res;
    }
}
